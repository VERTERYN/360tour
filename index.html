<!DOCTYPE html>

<html lang="nl">

<head>

  <meta charset="UTF-8">

  <meta name="viewport" content="width=device-width, initial-scale=1">

  <title>Virtuele Rondleiding</title>

  <!-- Pannellum CSS & JS via CDN -->

  <link

    rel="stylesheet"

    href="https://cdn.jsdelivr.net/npm/pannellum@2.5.6/build/pannellum.css"

  />

  <script src="https://cdn.jsdelivr.net/npm/pannellum@2.5.6/build/pannellum.js"></script>

  <style>

    html, body { margin:0; height:100%; }

    #panorama { width:100%; height:100%; }

  </style>

</head>

<body>

  <div id="panorama"></div>


  <script>

    // Aantal scènes

    const totalScenes = 27;

    // Object om al je scènes in te stoppen

    const scenesConfig = {};


    for (let i = 1; i <= totalScenes; i++) {

      const id = `scene${i}`;

      // Hotspots: Vorige én Volgende (behalve eerste/laatste)

      const hotSpots = [];

      if (i > 1) {

        hotSpots.push({

          pitch:  0,    // omhoog/omlaag: pas aan naar behoefte

          yaw:    90,   // draaiing: pas aan zodat de pijl in de juiste richting wijst

          type:  "scene",

          text:  "← Vorige",

          sceneId: `scene${i-1}`

        });

      }

      if (i < totalScenes) {

        hotSpots.push({

          pitch:  0,

          yaw:   -90,

          type:  "scene",

          text:  "Volgende →",

          sceneId: `scene${i+1}`

        });

      }


      scenesConfig[id] = {

        title:    `Scene ${i}`,

        type:     "equirectangular",

        panorama: `images/${id}.jpg`,

        hotSpots: hotSpots

      };

    }


    // Start de viewer

    pannellum.viewer('panorama', {

      default: {

        firstScene:        "scene1",

        sceneFadeDuration: 1000,

        autoLoad:          true

      },

      scenes: scenesConfig

    });

  </script>

</body>

</html>
